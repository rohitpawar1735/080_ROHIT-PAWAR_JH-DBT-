CREATE TABLE TEMPP(
S1 VARCHAR(30),
S2 VARCHAR(30),
MESSAGE VARCHAR(30)
);

DELIMITER //
CREATE PROCEDURE PROC1()
BEGIN
DECLARE X VARCHAR(30);
DECLARE Y VARCHAR(30);
DECLARE A VARCHAR(30);
DECLARE B VARCHAR(30);
SET X=CDAC
SET Y=DAC;
IF 
WHERE X LIKE '%Y%' THEN 
SET A='STRING IS PRESENT';
INSERT INTO TEMPP VALUES(X,Y,A)
ELSE
SET B='STRING IS NOT PRESENT'
INSERT INTO TEMPP VALUES(X,Y,B);
END;//
DELIMITER ;

CALL PROC1();

SELECT * FROM TEMPP;
